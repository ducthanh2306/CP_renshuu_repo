
#include<bits/stdc++.h> 
using namespace std;
 
int main(){
    int t;
    cin >> t;
    while(t--){
        int n;
        cin >> n;
        vector<int> a(n),per(n);
        unordered_map<int,int> mp;
        for(int i = 0 ; i < n;i++){
            cin >> per[i];
            mp[per[i]] = i;
        }
        for(int i = 0 ; i < n;i++){
            cin >> a[i];
        }
        int ok = 1;
        int idx = mp[a[0]];
        for(int i = 1; i < n;i++){
            int newIdx = mp[a[i]]; // root idx o trong mang per
            if(newIdx < idx) {
                ok = 0;
                break;
            }
            idx = newIdx;
            
        }
        if(ok) cout << "YES";
        else cout << "NO";
        cout <<"\n";
        
    }
}
